# UNAFFILIATED CARGO JOBS
# Basic cargo jobs to gauge the player's interest in the merchant lifestyle.

mission "Experience Cargo [0]"
	name "(Lv. 0+) Small delivery to <planet>"
	job
	repeat
	description "Deliver <cargo> to <destination>. Payment is <payment>."
	cargo random 75 125
	to offer
		not "Apoxys Level 10: declined"
	source
		government "Republic" "Free Worlds" "Syndicate" "Neutral" "Independent"
	destination
		distance 2 10
		government "Republic" "Free Worlds" "Syndicate" "Neutral" "Independent"
	on offer
		require "Apoxys Core DX"
	on visit
		dialog phrase "generic cargo on visit"
	on complete
		payment
		"combat rating" += 4
		"apoxys merchant interest" ++
		dialog phrase "generic cargo delivery payment"

# LEVEL 5 MERCHANT GUILD MISSIONS
# These are the first missions that give merchant rating, and as such, their plain CR rewards are slightly lower than would be proportional for typical experience jobs.

mission "Merchant's Guild Cargo [0]"
	name "(Lv. 5+) Small delivery to <planet>"
	job
	repeat
	description "Deliver <cargo> to <destination>. Payment is <payment>."
	cargo random 150 250
	to offer
		has "Apoxys Level 5: declined"
		not "Apoxys Level 15: declined"
		has "Merchant's Guild Invitation: done"
	source
		government "Republic" "Free Worlds" "Syndicate" "Neutral" "Independent"
	destination
		distance 2 10
		government "Republic" "Free Worlds" "Syndicate" "Neutral" "Independent"
	on offer
		require "Apoxys Core DX"
	on visit
		dialog phrase "generic cargo on visit"
	on complete
		payment
		"combat rating" += 12
		"merchant rating" += 12
		dialog phrase "generic cargo delivery payment"
